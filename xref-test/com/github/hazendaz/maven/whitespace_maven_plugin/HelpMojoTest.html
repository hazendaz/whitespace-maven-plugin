37      /**
38       * Execute covers invalid line and indent values and detailed output.
39       */
40      @Test
41      void executeWithInvalidConfiguration() {
42          HelpMojo mojo = new HelpMojo();
43          Log mockLog = mock(Log.class);
44          when(mockLog.isInfoEnabled()).thenReturn(true);
45          mojo.setLog(mockLog);
46          Whitebox.setInternalState(mojo, "lineLength", 0);
47          Whitebox.setInternalState(mojo, "indentSize", 0);
48          Whitebox.setInternalState(mojo, "detail", true);
49  
50          assertDoesNotThrow(mojo::execute);
51  
52          verify(mockLog).warn(contains("lineLength"));
53          verify(mockLog).warn(contains("indentSize"));
54          verify(mockLog).info(contains("Available parameters:"));
55          verify(mockLog).info(contains("User property: encoding"));
56      }

59       * Execute with a specific goal should emit only that goal output.

61      @Test
62      void executeWithSpecificGoal() {
63          HelpMojo mojo = new HelpMojo();
64          Log mockLog = mock(Log.class);
65          when(mockLog.isInfoEnabled()).thenReturn(true);
66          mojo.setLog(mockLog);
67          Whitebox.setInternalState(mojo, "goal", "verify");
68          Whitebox.setInternalState(mojo, "detail", false);
69  
70          assertDoesNotThrow(mojo::execute);

72          verify(mockLog).info(contains("whitespace:verify"));

75      /**
76       * getPropertyFromExpression handles valid and invalid expressions.
77       */
78      @Test
79      void getPropertyFromExpression() throws Exception {
80          assertEquals("some.property", invokePrivateStaticString("getPropertyFromExpression", "${some.property}"));
81          assertNull(invokePrivateStaticString("getPropertyFromExpression", "some.property"));
82          assertNull(invokePrivateStaticString("getPropertyFromExpression", "${some.${property}}"));
83      }

85      /**
86       * toLines and getIndentLevel handle wrapping and indentation details.
87       */
88      @SuppressWarnings("unchecked")
89      @Test
90      void toLinesAndIndentLevel() throws Exception {
91          Method toLines = HelpMojo.class.getDeclaredMethod("toLines", String.class, int.class, int.class, int.class);
92          toLines.setAccessible(true);
93          List<String> lines = (List<String>) toLines.invoke(null, "\tabc def ghi\u00A0jkl", 1, 2, 10);
94          assertNotNull(lines);
95          assertTrue(lines.size() >= 2);
96  
97          Method getIndentLevel = HelpMojo.class.getDeclaredMethod("getIndentLevel", String.class);
98          getIndentLevel.setAccessible(true);
99          int indent = (int) getIndentLevel.invoke(null, "\t\tvalue");
100         assertTrue(indent >= 2);
101     }

103     /**
104      * getSingleChild and findSingleChild throw on duplicates and missing child elements.
105      */
106     @Test
107     void xmlChildHelpersErrorPaths() throws Exception {
108         Document duplicateDoc = buildXml("<root><a>1</a><a>2</a></root>");
109         Document missingDoc = buildXml("<root></root>");
110 
111         assertThrows(MojoExecutionException.class, () -> invokePrivateNode("getSingleChild", duplicateDoc, "a"));
112         assertThrows(MojoExecutionException.class, () -> invokePrivateNode("getSingleChild", missingDoc, "a"));
113         assertThrows(MojoExecutionException.class, () -> invokePrivateNode("findSingleChild", duplicateDoc, "a"));
114         assertNull(invokePrivateNode("findSingleChild", missingDoc, "a"));
115     }
116 
117     private static Document buildXml(String xml) throws Exception {
118         return DocumentBuilderFactory.newInstance().newDocumentBuilder()
119                 .parse(new java.io.ByteArrayInputStream(xml.getBytes(StandardCharsets.UTF_8)));
120     }
121 
122     private static String invokePrivateStaticString(String methodName, String argument) throws Exception {
123         Method method = HelpMojo.class.getDeclaredMethod(methodName, String.class);
124         method.setAccessible(true);
125         return (String) method.invoke(null, argument);
126     }
127 
128     private static Node invokePrivateNode(String methodName, Document doc, String childName) throws Exception {
129         Method method = HelpMojo.class.getDeclaredMethod(methodName, Node.class, String.class);
130         method.setAccessible(true);
131         try {
132             return (Node) method.invoke(null, doc.getDocumentElement(), childName);
133         } catch (InvocationTargetException ex) {
134             Throwable cause = ex.getCause();
135             if (cause instanceof Exception) {
136                 throw (Exception) cause;
137             }
138             throw ex;
139         }
140     }
