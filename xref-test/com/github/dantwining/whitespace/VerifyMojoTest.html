15  package com.github.dantwining.whitespace;
16  
17  import java.io.File;
18  
19  import org.apache.maven.plugin.MojoExecutionException;
20  import org.apache.maven.plugin.MojoFailureException;
21  import org.junit.jupiter.api.Assertions;
22  import org.junit.jupiter.api.Test;
23  import org.powermock.reflect.Whitebox;
24  
25  /**
26   * The Class VerifyMojoTest.
27   */
28  public class VerifyMojoTest {
29  
30      /** The mojo. */
31      private VerifyMojo mojo;
32  
33      /**
34       * Verify test will pass due to no trailing spaces.
35       *
36       * @throws MojoExecutionException
37       *             the mojo execution exception
38       * @throws MojoFailureException
39       *             the mojo failure exception
40       */
41      @Test
42      void execute() throws MojoExecutionException, MojoFailureException {
43          mojo = new VerifyMojo();
44          Whitebox.setInternalState(mojo, "projectBasedir", new File("target/test-classes/verify"));
45          Whitebox.setInternalState(mojo, "extensions", "properties");
46          Whitebox.setInternalState(mojo, "encoding", "UTF-8");
47          mojo.execute();
48      }
49  
50      /**
51       * Verify test will fail due to trailing spaces.
52       *
53       * @throws MojoExecutionException
54       *             the mojo execution exception
55       * @throws MojoFailureException
56       *             the mojo failure exception
57       */
58      @Test
59      void executeFailureTest() throws MojoExecutionException, MojoFailureException {
60          mojo = new VerifyMojo();
61          Whitebox.setInternalState(mojo, "projectBasedir", new File("target/test-classes/verify"));
62          Whitebox.setInternalState(mojo, "extensions", "xml");
63          Whitebox.setInternalState(mojo, "encoding", "UTF-8");
64          Assertions.assertThrows(MojoFailureException.class, () -> {
65              mojo.execute();
66          });
67      }
68  
69  }
